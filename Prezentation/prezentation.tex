\section{Overwiew and goals}

\begin{frame}[fragile=singleslide]\frametitle{Overview}
\begin{itemize}
\item Drones collect aerial data on crops which gets sent to the server and analyzed by an AI algorithm
\item Detects pests, diseases, nutrient deficiencies, and irrigation needs.
\item Provides actionable recommendations to farmers.
\end{itemize}
\end{frame}

\begin{frame}[fragile=singleslide]\frametitle{Goals}
\begin{itemize}
\item Optimize precision agriculture in the Slovak Republic.
\item Increase crop yields by \textbf{15–25\%}.
\item Reduce pesticide use and promote sustainable farming practices.
\end{itemize}
\end{frame}



\section{Realization}

\begin{frame}[fragile=singleslide]\frametitle{System Architecture}
\begin{itemize}
  \item Drone platform: DJI Agriculture, PrecisionHawk, and other modern UAVs.
  \item Real-time image recognition and predictive analytics AI system.
  \item Combines multispectral drone imaging, weather data, and satellite imagery.
  \item Onboard AI for fast field-level analysis.
\end{itemize}
\end{frame}

\section{Impact}

\begin{frame}[fragile=singleslide]\frametitle{Market and Impact}
