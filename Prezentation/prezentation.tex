\section{Overwiew and goals}

\begin{frame}[fragile=singleslide]\frametitle{Overview}
\begin{itemize}
\item Drones collect aerial data on crops which gets sent to the server and analyzed by an AI algorithm
\item Decets pests, diseases, nutrient deficiencies, and irrigation needs.
\item Provides actionable recommendations to farmers.
\end{itemize}
\end{frame}

\begin{frame}[fragile=singleslide]\frametitle{Goals}
\begin{itemize}
\item Optimize precision agriculture in the Slovak Republic.
\item Increase crop yields by \textbf{15–25\%}.
\item Reduce pesticide use and promote sustainable farming practices.
\end{itemize}
\end{frame}



\section{Realization}

\begin{frame}[fragile=singleslide]\frametitle{System Architecture}
\begin{itemize}
  \item Drone platform : DJI Agriculture, PrecisionHawk, and other modern UAVs.
  \item Real-time image recognition and predictive analytics AI system.
  \item Combines multispectral drone imaging, weather data, and satellite imagery.
  \item Onboard AI for fast field-level analysis.
\end{itemize}
\end{frame}

\section{Critical Analysis}
\begin{frame}[fragile=singleslide]\frametitle{Comparison and Limitations}
\begin{itemize}
  \item \textbf{Compared to satellite monitoring:} Drones offer higher resolution and real-time data, but cover smaller areas.
  \item \textbf{Compared to manual inspection:} Faster and more consistent, but requires technical setup and maintenance.
  \item \textbf{Limitations:}
    \begin{itemize}
      \item Weather-dependent — drones can't fly in heavy rain or strong wind.
      \item Battery life limits flight time and coverage.
      \item Initial cost of equipment and training.
      \item AI accuracy depends on quality and diversity of training data.
    \end{itemize}
\end{itemize}
\end{frame}
